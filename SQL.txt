-- create tables
DROP TABLE IF EXISTS STUDENT CASCADE
DROP TABLE IF EXISTS ROOM CASCADE
DROP TABLE IF EXISTS ACTIVITY CASCADE
DROP TABLE IF EXISTS PARTICIPATE CASCADE
DROP TABLE IF EXISTS COMMENTPLANTFORM CASCADE

CREATE TABLE STUDENT (
	STU_ID CHAR(10) NOT NULL UNIQUE,
	STU_NAME VARCHAR(50) NOT NULL,
	STU_PASSWORD VARCHAR(20) NOT NULL,
	PRIMARY KEY (STU_ID)
);

CREATE TABLE ROOM (
	ROO_ID VARCHAR(10) NOT NULL UNIQUE,
	ROO_NAME VARCHAR(50) NOT NULL UNIQUE,
	ROO_CAPACITY integer NOT NULL,
	PRIMARY KEY (ROO_ID)
);
	

CREATE TABLE ACTIVITY (
	ACT_ID serial NOT NULL UNIQUE,
	ACT_NAME VARCHAR(50) NOT NULL,
	STU_ID CHAR(10) NOT NULL,
	ACT_STARTTIME TIMESTAMP NOT NULL,
	ACT_ENDTIME TIMESTAMP NOT NULL,
	ROO_ID VARCHAR(10) NOT NULL,
	ACT_CAPACITY INTEGER NOT NULL,
	CURRENTNUM INTEGER NOT NULL,
	PRIMARY KEY (ACT_ID),
	FOREIGN KEY (STU_ID) REFERENCES STUDENT (STU_ID) ON DELETE CASCADE ON UPDATE CASCADE,
	FOREIGN KEY (ROO_ID) REFERENCES ROOM (ROO_ID) ON DELETE CASCADE ON UPDATE CASCADE
);

CREATE TABLE PARTICIPATE (
	PAR_ID serial NOT NULL UNIQUE,
	ACT_ID serial NOT NULL,
	STU_ID CHAR(10) NOT NULL,
	PAR_TIME TIMESTAMP NOT NULL,
	PRIMARY KEY (PAR_ID),
	FOREIGN KEY (STU_ID) REFERENCES STUDENT (STU_ID) ON DELETE CASCADE ON UPDATE CASCADE,
	FOREIGN KEY (ACT_ID) REFERENCES ACTIVITY (ACT_ID) ON DELETE CASCADE ON UPDATE CASCADE
);
	
CREATE TABLE COMMENTPLATFORM (
	COM_ID serial NOT NULL UNIQUE,
	ACT_ID serial NOT NULL,
	STU_ID CHAR(10) NOT NULL,
	COM_CONTENT TEXT,
	COM_TIME TIMESTAMP NOT NULL,
	PRIMARY KEY (COM_ID),
	FOREIGN KEY (STU_ID) REFERENCES STUDENT (STU_ID) ON DELETE CASCADE ON UPDATE CASCADE,
	FOREIGN KEY (ACT_ID) REFERENCES ACTIVITY (ACT_ID) ON DELETE CASCADE ON UPDATE CASCADE
);
